    <script>
        // 1. Senarai kelas (Sediakan data kelas dahulu)
        const senaraiKelas = [
            "3 Inovatif", "3 Kreatif", 
            "4 Inovatif", "4 Kreatif", 
            "5 Inovatif", "5 Kreatif", 
            "6 Inovatif", "6 Kreatif", 
            "Guru", "Pelawat"
        ];

        const gameContext = "skpte"; 

        // 2. Bina kotak (kad) jadual di skrin DAHULU (Ditambah kolum Cuba)
        const gridContainer = document.getElementById('ranking-grid');
        senaraiKelas.forEach(kelas => {
            const safeId = kelas.replace(/\s+/g, '-'); 
            gridContainer.innerHTML += `
                <div class="class-card">
                    <h2 class="class-title">${kelas}</h2>
                    <table>
                        <thead>
                            <tr>
                                <th width="15%">Top</th>
                                <th width="15%">Cuba</th>
                                <th width="45%">Nama</th>
                                <th width="25%">Markah</th>
                            </tr>
                        </thead>
                        <tbody id="tbody-${safeId}">
                            <tr><td colspan="4" style="color:#666;"><i>Memuat turun data...</i></td></tr>
                        </tbody>
                    </table>
                </div>
            `;
        });

        // 3. Sambung ke Firebase & Tarik Data (SELEPAS UI dibina)
        try {
            const firebaseConfig = {
              apiKey: "AIzaSyCi9OJDWllc3YG-6loSE_NAPW_XFL1rAko",
              authDomain: "sifir-pro-leaderboard.firebaseapp.com",
              projectId: "sifir-pro-leaderboard",
              storageBucket: "sifir-pro-leaderboard.appspot.com",
              messagingSenderId: "877152351941",
              appId: "1:877152351941:web:101922b746e6e6173b22ac"
            };
            
            // Elak ralat jika firebase di-initialize lebih dari sekali
            if (!firebase.apps.length) {
                firebase.initializeApp(firebaseConfig);
            }
            
            const db = firebase.firestore();

            db.collection("scores")
              .where("context", "==", gameContext)
              .onSnapshot((querySnapshot) => {
                  let allScores = [];
                  querySnapshot.forEach((doc) => {
                      allScores.push(doc.data());
                  });

                  // Susun markah dari Paling Tinggi ke Rendah
                  allScores.sort((a, b) => b.score - a.score);

                  // Kumpulkan data mengikut kelas
                  let groupedByClass = {};
                  senaraiKelas.forEach(k => groupedByClass[k] = []);

                  allScores.forEach(s => {
                      if(groupedByClass[s.kelas] !== undefined) {
                          // Elak duplicate nama untuk murid yang sama (sebagai pelan sandaran)
                          if(!groupedByClass[s.kelas].find(murid => murid.name === s.name)) {
                              groupedByClass[s.kelas].push(s);
                          }
                      }
                  });

                  // Masukkan data Top 10 ke dalam jadual
                  senaraiKelas.forEach(kelas => {
                      const safeId = kelas.replace(/\s+/g, '-');
                      const tbody = document.getElementById(`tbody-${safeId}`);
                      
                      if (tbody) {
                          tbody.innerHTML = ''; 
                          const top10 = groupedByClass[kelas].slice(0, 10); 

                          if (top10.length === 0) {
                              tbody.innerHTML = '<tr><td colspan="4" style="color:gray;">Belum ada rekod</td></tr>';
                          } else {
                              top10.forEach((data, index) => {
                                  let rankDisplay = "";
                                  let rankClass = "";

                                  if (index === 0) { rankDisplay = "🥇 1"; rankClass = "rank-1"; }
                                  else if (index === 1) { rankDisplay = "🥈 2"; rankClass = "rank-2"; }
                                  else if (index === 2) { rankDisplay = "🥉 3"; rankClass = "rank-3"; }
                                  else { rankDisplay = index + 1; rankClass = "rank-other"; }
                                  
                                  // Tangkap nilai 'attempts', default kepada 1 jika tiada
                                  const attemptsDisplay = data.attempts || 1;

                                  tbody.innerHTML += `
                                      <tr class="${rankClass}">
                                          <td>${rankDisplay}</td>
                                          <td>${attemptsDisplay}</td>
                                          <td class="nama-murid">${data.name || "Tanpa Nama"}</td>
                                          <td>${data.score || 0}</td>
                                      </tr>
                                  `;
                              });
                          }
                      }
                  });
              }, (error) => {
                  console.error("Gagal menarik data dari Firebase:", error);
              });

        } catch (error) {
            console.error("Masalah sambungan ke Firebase:", error);
        }
